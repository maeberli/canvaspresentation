Contexte graphique

L'interaction avec le contexte graphique est faite en JavaScript.

  • Formes de base à disposition:
    carré, cercle, secteur de cercle, texte...
  • Styles et couleurs du pinceau associé au contexte graphique: largeur, couleur de fond, ...
  • Utilisation semblable au contexte graphique de Java, Qt, etc.

Filter

function filterGray() {	
  if (canvas.width > 0 && canvas.height > 0) {
    var imageData = c.getImageData(0, 0, 
	                            canvas.width, canvas.height);
	
    var d = imageData.data;
    for (var i = 0; i < d.length; i += 4) {
      var r = d[i];
      var g = d[i + 1];
      var b = d[i + 2];
      d[i] = d[i + 1] = d[i + 2] = (r+g+b)/3;
    }    
	
    c.putImageData(imageData, 0, 0);	
  }
}

function drawPoint( point, title ) {
  var location = "( " + point.x + ", " + point.y + " )";
  var titleWidth = c.measureText( title );
  var locationWidth = c.measureText( location );
  
  // Set font style for current graphic context.
  c.font = "12px sans-serif";
  
  // Display the title and the location.
  c.fillText( title, point.x-12-titleWidth.width, point.y + 12 );
  c.fillText( location , point.x-12-locationWidth.width, point.y + 24 );
  
  // Draw a circle at the specific position.
  c.beginPath();
  c.arc( point.x, point.y, 5, 0, Math.PI*2, true );
  c.fill();
}

function drawBezierStep( pA, pB, pC, t ) {
  var vecAB = new Vector( pB.x-pA.x, pB.y-pA.y );
  var vecBC = new Vector( pC.x-pB.x, pC.y-pB.y );
  var pQ1 = new Vector( pA.x + vecAB.x * t, pA.y + vecAB.y * t );
  var pQ2 = new Vector( pB.x + vecBC.x * t, pB.y + vecBC.y * t );
  var vecQ1Q2 = new Vector( pQ2.x-pQ1.x, pQ2.y-pQ1.y );
  var pQ = new Vector( pQ1.x + vecQ1Q2.x*t, pQ1.y + vecQ1Q2.y*t );
  
  c.beginPath();
  c.arc( pQ1.x, pQ1.y, 2, 0 , Math.PI*2, true );
  c.arc( pQ2.x, pQ2.y, 2, 0 , Math.PI*2, true );
  c.fill();
  
  c.beginPath();
  c.moveTo( pQ1.x, pQ1.y );
  c.lineTo( pQ2.x, pQ2.y );
  c.stroke();
  
  c.beginPath();  
  c.arc( pQ.x, pQ.y, 3, 0, Math.PI*2, true );
  c.fill();
}

// New global variable, which holds the current animation step.
var step = 0;
// method which is called periodically
function draw(){
  if( step <= 10 ) {
    drawBezierStep( pA, pB, pC, 1/10 * step );
    step += 1;
  } else {
    step = 0;
    initScene();
  }
}

function initContext() {
  canvas = document.getElementById( 'myCanvas' );
  c = canvas.getContext( '2d' );

  pA = new Vector( 60, 60 );
  pB = new Vector( 70, 170 );
  pC = new Vector( 380, 170);
  
  initScene();

  return setInterval( draw, 500 );
}

function initScene() {
  c.clearRect( 0, 0, canvas.width, canvas.height );
  drawPoint( pA, "pA" );
  drawPoint( pB, "pB" );
  drawPoint( pC, "pC" );
  drawBezierCurve();
}

Simple KineticJS exemple

var stage = new Kinetic.Stage({
  container: 'container', width: 400, height: 120
});
var layerAnim = new Kinetic.Layer();

var centerX = stage.getWidth() / 2;
var centerY = stage.getHeight() / 2;

var hexagon = new Kinetic.RegularPolygon({
  x: centerX, y: centerY,
  sides: 6, radius: 55, strokeWidth: 6
});

layerAnim.add(hexagon);
stage.add(layerAnim);

var anim = new Kinetic.Animation(function(frame) {
  hexagon.setX(90 * Math.sin(frame.time*2*Math.PI/2000) + centerX);
}, layerAnim);
anim.start();
